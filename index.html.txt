function createPuzzle(containerId, image, callback) {
  let container = document.getElementById(containerId);
  let order = [...Array(9).keys()].sort(() => Math.random() - 0.5);

  let selected = null;

  order.forEach(i => {
    let tile = document.createElement("div");
    tile.className = "tile";
    tile.style.backgroundImage = `url(${image})`;
    tile.style.backgroundPosition = `${-(i%3)*100}px ${-Math.floor(i/3)*100}px`;
    tile.dataset.correct = i;
    tile.dataset.current = order.indexOf(i);

    tile.onclick = () => {
      if (!selected) {
        selected = tile;
        tile.style.opacity = "0.5";
      } else {
        let temp = selected.style.backgroundPosition;
        selected.style.backgroundPosition = tile.style.backgroundPosition;
        tile.style.backgroundPosition = temp;

        selected.style.opacity = "1";
        selected = null;

        checkSolved(container, callback);
      }
    };

    container.appendChild(tile);
  });
}

function checkSolved(container, callback) {
  let tiles = container.children;
  let correct = true;

  for (let i = 0; i < tiles.length; i++) {
    let x = -(i%3)*100 + "px";
    let y = -Math.floor(i/3)*100 + "px";
    if (tiles[i].style.backgroundPosition !== x + " " + y) {
      correct = false;
      break;
    }
  }

  if (correct) callback();
}
